    # Input 输入框

    # 基本用法

    点击查看代码 
      <cf-input placeholder="请输入内容" type="text"></cf-input>
    1

    # 绑定事件

    点击查看代码 
          <cf-input placeholder="请输入内容" type="text" v-model="username"></cf-input>
    1

    # 清空内容

    点击查看代码 
      <cf-input placeholder="请输入用户名" type="text" v-model="username" clearable></cf-input>
    1

    # 显示密码

    点击查看代码 
      <cf-input placeholder="请输入密码" type="password" v-model="password" showPassword></cf-input>
    1

    # 禁用

    点击查看代码 
      <cf-input placeholder="请输入内容" type="password" name="password" disabled></cf-input>
    1
